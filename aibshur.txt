<script>
function openForm(type){

    document.getElementById("home").style.display = "none";

    let title = "";
    let form = "";

    if(type === "id"){
        title = "تقديم هوية";
        form = `
        <input id="name" placeholder="الاسم">
        <input id="age" placeholder="العمر">
        `;
    }

    if(type === "drive"){
        title = "رخصة قيادة";
        form = `
        <input id="name" placeholder="الاسم">
        <input id="exp" placeholder="خبرة القيادة">
        `;
    }

    if(type === "weapon"){
        title = "رخصة سلاح";
        form = `
        <input id="name" placeholder="الاسم">
        <input id="reason" placeholder="سبب الطلب">
        `;
    }

    if(type === "report"){
        title = "بلاغ للشرطة";
        form = `
        <input id="name" placeholder="اسمك">
        <input id="details" placeholder="تفاصيل البلاغ">
        `;
    }

    form += `<button onclick="send('${title}')">إرسال 🚀</button>`;

    document.getElementById("formBox").innerHTML = `<h3>${title}</h3>` + form;
    document.getElementById("formBox").style.display = "block";
}

function send(title){

    const webhook = "https://discord.com/api/webhooks/1492085932109856859/WA6fsAV9NByRAd57Unj4GZLLbrYHXgr0SCCjK-UObKH9q7NqGgFq-qW7UiWGqM_iv1nU";

    let inputs = document.querySelectorAll("#formBox input");

    for (let i of inputs){
        if(!i.value){
            alert("❌ عبي كل البيانات");
            return;
        }
    }

    let content = "📩 طلب جديد\n📂 " + title + "\n\n";

    inputs.forEach(i=>{
        content += "• " + i.placeholder + ": " + i.value + "\n";
    });

    fetch(webhook,{
        method:"POST",
        headers:{"Content-Type":"application/json"},
        body: JSON.stringify({content})
    })
    .then(()=>{
        alert("✅ تم الإرسال بنجاح");
        location.reload();
    })
    .catch(()=>{
        alert("❌ فشل الإرسال");
    });
}
</script>
